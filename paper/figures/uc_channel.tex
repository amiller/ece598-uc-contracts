\begin{bbox}[title={Functionality $\F_{channel}(p_r,p_s)$}]

\OnInput \inmsg{m} from $p_r$:
	\begin{renumerate}
		\item Store $m$ and leak $m \rightarrow \mathcal{A}$
	\end{renumerate}
	
\OnInput \inmsg{\texttt{ok}} from $\mathcal{A}$:
	\begin{renumerate}
		\item Send $m \rightarrow p_s$
	\end{renumerate}
\end{bbox}
