\begin{bbox}[title={$\mathcal{F}_\mathsf{async}^{P_i, P_j}$}]

\OnInput $x$ from $P_i$:

	\quad Send (\textsc{Eventually}, \textsf{send}, $x$, $|x|$) to $\mathcal{W}$
	
\OnInput (\textsf{send}, $x$) from $\mathcal{W}$:

	\quad Run \textsf{send}($x$)
	
Function \textsf{send}($x$):

	\quad Send $x$ to $P_j$
\end{bbox}
